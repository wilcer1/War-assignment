    def read_highscore(self):

    

  

  

  

    

      

        #
      

      
    

    

      
        highscore_list = []
        with open(self.path, "r") as text_file:
            for line in text_file:
                line = line.rstrip("\n")
                temp = line.split(":")
                score = int(temp[0])
                name = temp[1]
                highscore_list.append((score, name))

        sorted_highscore_list = self.sort_highscore(highscore_list)
        return sorted_highscore_list
